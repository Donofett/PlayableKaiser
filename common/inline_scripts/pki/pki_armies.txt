allow = {
    custom_tooltip_success = {
        text = pki_$ARMY$_army_limit
        hidden_trigger = {
            event_target:pki_player = {
                check_variable = {
                    which = pki_$ARMY$_current
                    value < value:pki_army_$ARMY$_limit
                }
            }
        }
    }
    custom_tooltip_fail = {
        text = pki_$ARMY$_reached_limit
        hidden_trigger = {
            event_target:pki_player = {
                check_variable = {
                    which = pki_$ARMY$_current
                    value < value:pki_army_$ARMY$_limit
                }
            }
        }
    }
}

on_queued = {
    event_target:pki_player = {
        change_variable = {
            which = pki_$ARMY$_current
            value = 1
        }
    }
}

on_unqueued = {
    event_target:pki_player = {
        change_variable = {
            which = pki_$ARMY$_current
            value = -1
        }
    }
}
