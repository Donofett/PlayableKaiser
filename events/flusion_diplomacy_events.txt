 # fromfrom / event_target:solakia_flunat = Solakia
  country_event = {
	id = flunat.0101
	title = flunat.0101.name
	desc = flunat.0101a.desc
	desc = flunat.0101b.desc
	desc = flunat.0101c.desc
	desc = flunat.0101d.desc
	desc = flunat.0101e.desc
	desc = flunat.0101f.desc

	is_triggered_only = yes
	diplomatic = yes
	picture_event_data = {
		portrait = event_target:dofus_flunat
		planet_background = root.owner
		graphical_culture = humanoid_01
		city_level = root.owner
		room = personality_federation_builders_room
	}
	location = event_target:flusionhome

	trigger = { #for events which inherit from this one
		NOT = {
			has_global_flag = solakia_conquered
		}
	}

	option = { # Lore
		name = flunat.0101.a
		custom_tooltip = flunat_get_lore
		is_dialog_only = yes
		response_text = flunat.0101.a.response
	}

	option = { # Gib land pls
		name = flunat.0101.b
		custom_tooltip = flunat_demand_land
		is_dialog_only = yes
		response_text = flunat.0101.b.response
	}

	option = { # WAR
        name = flunat.0101.c
        custom_tooltip = flunat_war_dec
		trigger = {
            NOR = {
                has_country_flag = war_with_solakia 
				
            }
        }
        is_dialog_only = yes
        response_text = flunat.0101.c.response
        hidden_effect = {
			
			event_target:playerflusion = {
				change_variable = {
					which = solakians_con
					value = 1
				}
			}
			set_country_flag = war_with_solakia
			check_flusionian_war = yes
			random_situation = {
				limit = {
					is_situation_type = situation_world_tension
				}
				add_situation_progress = 1
			}
            start_situation = {
                type = situation_solakian_conquest
                target = root
                effect = {
                    set_situation_approach = approach_mixed
                }
            }
        }
    }

	option = { # Bye
		name = flunat.0101.d
		default_hide_option = yes
	}
}

#Solakian War Win
country_event = {
	id = flunat.0102
	title = flunat.0101.name
	desc = flunat.0102.desc

	is_triggered_only = yes
	diplomatic = yes
	picture_event_data = {
		portrait = event_target:dofus_flunat
		planet_background = root.owner
		graphical_culture = humanoid_01
		city_level = root.owner
		room = personality_federation_builders_room
	}
	location = event_target:flusionhome
	
	option = {
		name = flunat.0102.a
		hidden_effect = {
			event_target:playerflusion = {
				change_variable = {
					which = solakians_con
					value = 1
				}
			}
			remove_country_flag = war_with_solakia
			check_flusionian_war = yes
			set_country_flag = solak_pop_acquired
			event_target:solakia_flunat = {
				destroy_country = yes
			}
			random_situation = {
				limit = {
					is_situation_type = situation_world_tension
				}
				add_situation_progress = 5
			}
			random_situation = {
				limit = {
					is_situation_type = situation_solakian_conquest
				}
				destroy_situation = this
			}
			set_global_flag = solakia_conquered
			event_target:playerflusion = {
				add_deposit = d_manglarian_junkyard
				add_deposit = d_solakian_factory
				add_deposit = d_meowriana_trench
				add_deposit = d_altarian_jungle
				player_flusion_liberated_solak = yes
				while = {
					count = 5
					create_pop = {
						species = event_target:solakians
					}
					last_created_pop = {
						add_modifier = {
							modifier = "pop_recently_conquered"
							days = 3600
						}
					}
				}
			}
		}
	}
}
