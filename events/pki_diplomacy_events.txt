namespace = pki_flusion_nations

#Find the correct nation to diplomacy with
# Via on_custom_diplomacy | this = target country (player); from = country being contacted
country_event = {
	id = pki_flusion_nations.0000
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			from = { is_country_type = flusion_nation }
			pki_is_katzenartig_imperium = yes
		}
	}

	immediate = {
		set_country_flag = pki_contacted_country
		from = {
			switch = {
				trigger = has_country_flag
				inline_script = {
					script = pki/pki_inline_script
					SCRIPT_NAME = country_contact
				}
			}
		}
	}
}

country_event = {
	id = pki_flusion_nations.01
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		change_variable = {
			which = pki_building_slots
			value = event_target:pki_HAR_country.pki_building_slots
		}

		event_target:pki_HAR_country = {
			multiply_variable = {
				which = pki_industry_level
				value = 0.5
			}
		}


		change_variable = {
			which = pki_industry_level
			value = event_target:pki_HAR_country.pki_industry_level
		}

		while = {
			count = event_target:pki_HAR_country.pki_pops
			event_target:flusionhome = {
				create_pop = {
					species = event_target:pki_HAR_species
					ethos = random
					effect = {
						add_modifier = {
							modifier = pop_recently_conquered
							days = 3600
						}
					}
				}
			}
		}
	}
}
